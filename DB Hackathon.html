The objective was to complete a set of specified tasks within a one-hour time frame and submit the resulting SQL script, demonstrating proficiency in SQL skills. My Code


SELECT DISTINCT t.Tree AS CommonName, CASE
WHEN soil_count < 6 THEN 'Picky'
WHEN soil_count < 8 THEN 'Moderate'
ELSE 'Tolerant'
END AS PickinessCategory
FROM TREES t
JOIN TREEZONES tz ON t.TreeID = tz.TreeID
JOIN ZONES z ON tz.ZoneID = z.ZoneID
JOIN (
	SELECT TreeID, COUNT(DISTINCT TreeID || Soil) AS soil_count
	FROM SOILS
	GROUP BY TreeID
) s ON t.TreeID = s.TreeID
WHERE (
	SELECT COUNT(DISTINCT ZoneID)
	FROM TREEZONES
	WHERE TreeID = t.TreeID
) >= 7
ORDER BY CommonName;
