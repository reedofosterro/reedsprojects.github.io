Heart Disease Prevention and Correlation Analysis

Project Details

Worked in groups of 3.

Conducted a comprehensive analysis of heart disease prevalence using computational thinking and Python programming. Utilized a dataset sourced from Kaggle to examine the relationship between various clinical variables and the likelihood of developing heart disease. Developed graphical representations and performed linear regression analysis to identify key correlations. The project aims to provide actionable insights for individuals to mitigate their risk of heart disease.

import pandas as pd
import matplotlib.pyplot as plt

# Read the data from an Excel file into a pandas DataFrame
data = pd.read_csv('heart_disease_prediction.csv')

# Create a new DataFrame that only includes rows where Cholesterol is over 240
high_cholesterol = data[data['Cholesterol'] > 240]

# Group the data by Heart Disease and count the number of occurrences of each value
heart_disease_counts = high_cholesterol.groupby('Heart Disease')['Cholesterol'].count()

# Create a pie chart to show the correlation between high cholesterol levels and heart disease
plt.pie(heart_disease_counts, labels=heart_disease_counts.index, autopct='%1.1f%%')
plt.title('Correlation between high cholesterol and heart disease')

plt.savefig('cholesterol_contribution.png', dpi=300, bbox_inches='tight')
plt.show()

#GRAPH 2
def plotIdentityContribution():  
    # Load dataframe  
    df = pd.read_csv("heart_disease_prediction.csv")
  
    # map sex data  
    df['Sex'] = df['Sex'].map({0: 'Female', 1: 'Male'})   

    # Subset dataframe to only include instances where Heart Disease is 'Presence'  
    presence = df[df['Heart Disease'] == 'Presence']   

    # Create separate dataframes for males and females   
    males = presence[presence['Sex'] == 'Male']   
    females = presence[presence['Sex'] == 'Female']   

    # Group by age and sex to get the count of heart disease cases   
    grouped = presence.groupby(['Age', 'Sex'])['Heart Disease'].count().reset_index()   

    # Merge male and female counts into a single DataFrame   
    merged = grouped.pivot(index='Age', columns='Sex', values='Heart Disease').fillna(0)   

    # Create bar plot using the merged DataFrame   
    merged.plot(kind='bar', xlabel='Age', ylabel='Number of cases', title='Heart Disease by Age and Gender (Presence)')   
    plt.legend(title='Gender', loc='upper left')   
    plt.tight_layout()   
    plt.savefig('hd_age_gender_presence.png', dpi=300, bbox_inches='tight')   
    plt.show() 

plotIdentityContribution()
 

#GRAPH 3
def plotMaxHR():
    df = pd.read_csv('heart_disease_prediction.csv')

    # Filter the DataFrame to include only rows where Max HR is greater than or equal to 160
    df_160 = df[df['Max HR'] >= 160]

    # Group the filtered DataFrame by Heart Disease and count the number of rows for each group
    df_counts = df_160.groupby('Heart Disease').size()

    # Create a bar chart of the counts
    df_counts.plot(kind='bar')

    # Set the X-axis and Y-axis labels
    plt.xlabel('Heart Disease')
    plt.ylabel('Count')

    plt.title('Number of People with Max HR of 160 in terms of correlation with Heart Disease')
    plt.savefig('MaxHR.png', dpi=300, bbox_inches='tight')
    # Show the plot
    plt.show()

plotMaxHR() # Call the function to generate the plot

#LINEAR REGRESSION
# Load numeric data
num_data = pd.read_csv('heart_disease_prediction.csv')  

# Subset data to relevant columns
x = num_data[['Sex', 'Chest pain type', 'BP', 'Max HR', 'Exercise angina', 'Number of vessels fluro', 'Thallium']]  

# Extract target variable
y = num_data['Heart Disease'].map({'Absence': 0, 'Presence': 1})  

# Fit linear regression model
model = LinearRegression().fit(x,y)  

print('Coefficients: ', model.coef_)  
print('Intercept: ', model.intercept_)  
print('R-squared: ', model.score(x, y))  

# Add a constant term to the predictor variables  
x = sm.add_constant(x)  

# Create a linear regression model and fit it to the data  
model = sm.OLS(y, x).fit()  

# Print the summary of the regression results, including p-values for the coefficients  
print(model.summary())
